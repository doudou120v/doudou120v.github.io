function msToHMS(ms) {
  let seconds = Math.floor(ms / 1000);
  let hours = Math.floor(seconds / 3600);
  seconds %= 3600;
  let minutes = Math.floor(seconds / 60);
  seconds %= 60;
  return [hours, minutes, seconds].map(v => String(v).padStart(2, '0')).join(':');
}
